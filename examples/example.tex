\section{Introduction}
\subsection{Why flavour physics?}
\frame{
\frametitle{Why flavour physics?}
\begin{itemize}
\item Theoretical interest: why do fermions have a large range of masses ($m_t/m_e \approx 350000$)?
\item Flavour physics is (one of) the best places to look for New Physics. 
\begin{itemize}
\item Flavour Changing Neutral processes are heavily suppressed in the SM (loop and mass suppressed).
\item But New Physics might be not so suppressed: sizable contributions.
\end{itemize}
\item Experiments are capable of good sensitivities: LHCb, BaBar, Belle.
\end{itemize}
}


\subsection{Some $B$ physics anomalies}
\frame{
\frametitle{Some $B$ physics anomalies}
\begin{itemize}
\item Rare $B$ decays: $b\to s \mu^+ \mu^-$ and $b \to s e^+ e^-$:
\begin{itemize}
\item $R_{K^{(*)}} = \frac{\mathcal{B}(B\to K^{(*)}\mu^+ \mu^-)}{\mathcal{B}(B\to K^{(*)}e^+ e^-)}$.
\item  $R_K^\mathrm{SM} = 1.00 \pm 0.01$, \qquad $R_K^\mathrm{exp} = 0.745^{+0.090}_{-0.074}\pm 0.036$, \qquad (\alert{$2.6 \sigma$}). \footnote{R. Aaij \textit{et al} (LHCb Collaboration) arXiv:1406.6482}
\item  $R_{K^*}^\mathrm{SM} = 1.00 \pm 0.01$, \qquad $R_{K^*}^\mathrm{exp} = 0.685^{+0.113}_{-0.069}\pm 0.047$, \qquad (\alert{$2.5 \sigma$}). \footnote{S. Bifani. CERN Seminar, 18 April 2017 \& arXiv:1705.05802}
\item Angular observables $P_4'$, $P_5'$.
~\\~\\

\item Violation of \alert{Lepton Flavour Universality}?
\end{itemize}~\\~\\

\item $B_s$ mixing:
\begin{itemize}
\item $\Delta M_S^\mathrm{SM} = 20.01 \pm 1.25\,\mathrm{ps}^{-1}$, \qquad$\Delta M_S^\mathrm{exp} = 17.757 \pm 0.021 \,\mathrm{ps}^{-1}$, \quad (\alert{$1.8\sigma$}). \footnote{L. Di Luzio, M. Kirk, A. Lenz. arXiv:1712.06572}

\end{itemize}
\end{itemize}
}
